## Usage

```tsx
import {
  Tooltip,
  TooltipContent,
  TooltipTrigger,
} from "@/registry/ui/tooltip/tooltip";
```

```tsx
<Tooltip>
  <TooltipTrigger>Its Above Me!</TooltipTrigger>
  <TooltipContent>
    <p>Heyyo, I'm a tooltip!</p>
  </TooltipContent>
</Tooltip>
```
