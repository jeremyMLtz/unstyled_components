## Usage

```tsx
import {
  Card,
  CardContent,
  CardDescription,
  CardFooter,
  CardHeader,
  CardTitle,
} from "@/registry/ui/card/card";
```

```tsx
<Card>
  <CardHeader>
    <CardTitle>Setup App</CardTitle>
    <CardDescription>Lets get your app running!</CardDescription>
  </CardHeader>
  <CardContent>
    <span>This is some cool content</span>
  </CardContent>
  <CardFooter>
    <span>This is a footer</span>
  </CardFooter>
</Card>
```
